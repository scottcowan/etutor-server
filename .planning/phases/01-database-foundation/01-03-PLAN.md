    create_session(child_id: str, session: AsyncSession) -> SessionModel:
      model = SessionModel(id=str(uuid.uuid4()), child_id=child_id)
      session.add(model); await session.commit(); await session.refresh(model); return model

    get_session(session_id: str, session: AsyncSession) -> Optional[SessionModel]:
      result = await session.execute(select(SessionModel).where(SessionModel.id == session_id))
      return result.scalar_one_or_none()

    log_turn(child_id: str, question: str, answer: str, session: AsyncSession, *, topic: str = None, session_id: str = None) -> InteractionEventModel:
      model = InteractionEventModel(id=str(uuid.uuid4()), child_id=child_id, question=question, answer=answer, topic=topic, session_id=session_id)
      session.add(model); await session.commit(); await session.refresh(model); return model

    Signature note: topic and session_id are keyword-only (after *) to match the call in services/sessions.py where topic is already optional. This prevents positional-argument breakage.

    get_session_history(child_id: str, session: AsyncSession, limit: int = 50) -> list[InteractionEventModel]:
      result = await session.execute(
        select(InteractionEventModel)
        .where(InteractionEventModel.child_id == child_id)
        .order_by(InteractionEventModel.timestamp.asc())
        .limit(limit)
      )
      return list(result.scalars().all())

    create_or_get_mastery_state(child_id: str, kc_id: str, session: AsyncSession) -> MasteryStateModel:
      existing = await session.get(MasteryStateModel, (child_id, kc_id))
      if existing:
          return existing
      model = MasteryStateModel(child_id=child_id, kc_id=kc_id)
      session.add(model); await session.commit(); await session.refresh(model); return model

    update_mastery_state(child_id: str, kc_id: str, session: AsyncSession, **fields) -> None:
      model = await session.get(MasteryStateModel, (child_id, kc_id))
      if model is None:
          raise ValueError(f"MasteryState not found for ({child_id}, {kc_id})")
      for k, v in fields.items():
          setattr(model, k, v)
      model.updated_at = datetime.utcnow()
      await session.commit()

    get_session_history with limit: the most-recent limit turns. Use .order_by(timestamp.asc()).limit(limit) — this returns the oldest N of all turns, not the most recent. For the test "most recent 2 of 5" semantics, use a subquery or offset. Simpler: fetch all ordered ASC, return last limit items in Python: `return rows[-limit:]` after fetching. This avoids SQL complexity for Phase 1.
